;; This section defines simple functions used in Chapter 1
;; rotate-cw 
;; rotate-ccw
;; rotate-180

(define (rotate-cw x) 
  (rotate 90 x))
(define (rotate-ccw x)
  (rotate -90 x))
(define (rotate-180 x)
  (rotate 180 x))
